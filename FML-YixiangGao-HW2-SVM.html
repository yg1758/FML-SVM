sex_label = {'M': 1, "F": -1, "I": 0}

def label_sex(s):
    return sex_label[s]

def label_ring(r):
    if r >= 9:
        return 1
    else:
        return -1

df['Sex_label'] = df['Sex'].apply(lambda x: label_sex(x))
df['Rings_label'] = df['Rings'].apply(lambda x: label_ring(x))
